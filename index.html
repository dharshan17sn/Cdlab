%{
#include "y.tab.h"
%}
%%
"for"   { return FOR; }
"("     { return LPAREN; }
")"     { return RPAREN; }
"{"     { return LF; }
"}"     { return RF; }
"=="    { return EQ; }
"<="    { return LE; }
">="    { return GE; }
"+="    { return ADDEQ; }
"-="    { return SUBEQ; }
"++"    { return INC; }
"--"    { return DEC; }
"="     { return '='; }
"-"     { return '-'; }
"+"     { return '+'; }
">"     { return '>'; }
"<"     { return '<'; }
";"     { return ';'; }
[a-zA-Z]+  { return ALPH; }
[0-9]+     { return NUM; }
[ \t\n]    { /* ignore */ }
"#"        { return 0; }
.          { /* ignore */ }
%%
int yywrap(){ return 1; }

YACC Code

%{
#include 
#include 
int yylex();
int count = 0;
int yyerror();
%}
%token FOR LPAREN RPAREN LF RF ALPH NUM EQ LE GE ADDEQ SUBEQ INC DEC
%%
S : forlist ;
forlist : I | forlist I ;
I : FOR A B  { count++; } ;
A : LPAREN E ';' E ';' E RPAREN ;
E : ALPH Z NUM
  | ALPH Z ALPH
  | ALPH U
  | /* empty */
  ;
Z : '=' | '>' | '<' | LE | GE | EQ | ADDEQ | SUBEQ ;
U : INC | DEC ;
B : LF body RF ;
body : /* empty */
     | body I
     | body ALPH ';'
     ;
%%
int main(){
    printf("Enter for loop(s) followed by #:\n");
    yyparse();
    if(count < 3){
        printf("Invalid: minimum 3 nesting levels required (found %d)\n", count);
    } else {
        printf("Valid — number of FOR loops: %d\n", count);
    }
    return 0;
}
int yyerror(){
    printf("Invalid\n");
    exit(0);
}

Program 4 — Three Address Code (Quadruple & Triple Representation) LEX + YACC ▼

LEX Code (prog4.l)

%{
#include "y.tab.h"
extern YYSTYPE yylval;
%}
%%
[0-9]+     { yylval.sym = yytext[0]; return NUMBER; }
[a-zA-Z]+  { yylval.sym = yytext[0]; return LETTER; }
\n         { return 0; }
.          { return yytext[0]; }
%%
int yywrap(){ return 1; }

YACC Code (prog4.y)

%{
#include 
#include 
struct entry {
    char op, arg1, arg2, result;
} table[20];
int idx  = 0;
int flag = 0;
int tnum = 1;   /* temp counter: T1, T2, ... */
char newTemp(){
    return (char)('0' + tnum++);
}
char addEntry(char arg1, char arg2, char op){
    char t = newTemp();
    table[idx].op     = op;
    table[idx].arg1   = arg1;
    table[idx].arg2   = arg2;
    table[idx].result = t;
    idx++;
    return t;
}
%}
%union { char sym; }
%token  LETTER NUMBER
%type   expr
%left  '+' '-'
%right '*' '/'
%%
statement : LETTER '=' expr ';'
            {
                table[idx].op     = '=';
                table[idx].arg1   = $3;
                table[idx].arg2   = ' ';
                table[idx].result = $1;
                idx++;
            }
          | expr ';'
          ;
expr : expr '+' expr  { $$ = addEntry($1, $3, '+'); }
     | expr '-' expr  { $$ = addEntry($1, $3, '-'); }
     | expr '*' expr  { $$ = addEntry($1, $3, '*'); }
     | expr '/' expr  { $$ = addEntry($1, $3, '/'); }
     | '(' expr ')'   { $$ = $2; }
     | NUMBER          { $$ = $1; }
     | LETTER          { $$ = $1; }
     ;
%%
void printCode(){
    printf("\n=== Three-Address Code ===\n");
    for(int i = 0; i < idx; i++){
        if(table[i].op == '=')
            printf("%c = %c\n", table[i].result, table[i].arg1);
        else
            printf("T%d = %c %c %c\n", i+1, table[i].arg1, table[i].op, table[i].arg2);
    }
    printf("\n=== Quadruple (Index, Op, Arg1, Arg2, Result) ===\n");
    printf("%-6s %-5s %-5s %-5s %-6s\n","Index","Op","Arg1","Arg2","Result");
    for(int i = 0; i < idx; i++){
        if(table[i].op == '=')
            printf("%-6d %-5c %-5c %-5s %-6c\n", i, table[i].op, table[i].arg1, "-", table[i].result);
        else
            printf("%-6d %-5c %-5c %-5c T%-5d\n", i, table[i].op, table[i].arg1, table[i].arg2, i+1);
    }
    printf("\n=== Triple (Index, Op, Arg1, Arg2) ===\n");
    printf("%-6s %-5s %-5s %-5s\n","Index","Op","Arg1","Arg2");
    for(int i = 0; i < idx; i++)
        printf("%-6d %-5c %-5c %-5c\n", i, table[i].op, table[i].arg1, table[i].arg2);
}
int main(){
    printf("Enter expression (e.g.  a = b + c * d; ):\n");
    yyparse();
    if(!flag) printCode();
    return 0;
}
int yyerror(char *s){
    flag = 1;
    printf("Error: %s\n", s);
    return 0;
}

%{
#include 
#include 
#include 
void yyerror(const char *s);
int yylex();
%}
%union { char *id; int num; char *str; }
%token   ID
%token  NUM
%token  STRING
%token INT MAIN PRINTF ADD LPAREN RPAREN SEMI COMMA LBRACE RBRACE ASSIGN RETURN
%start program
%%
program
    : INT MAIN LPAREN RPAREN LBRACE
      {
          printf(".data\n");
          printf("  .LC0: .string \"Sum %%d\\n\"\n");
          printf(".text\n");
          printf("  .globl main\n");
          printf("main:\n");
          printf("  push %%ebp\n");
          printf("  movl %%esp, %%ebp\n");
      }
      stmt_list RBRACE
      {
          printf("  movl $0, %%eax\n");
          printf("  pop %%ebp\n");
          printf("  ret\n");
      }
    ;
stmt_list
    : stmt
    | stmt_list stmt
    ;
stmt
    : INT idlist SEMI
      { /* declarations — no code emitted */ }
    | ID ASSIGN NUM SEMI
      { printf("  movl $%d, -%s\n", $3, $1); }
    | ID ASSIGN ID ADD ID SEMI
      {
          printf("  movl -%s, %%eax\n", $3);
          printf("  addl -%s, %%eax\n", $5);
          printf("  movl %%eax, -%s\n", $1);
      }
    | PRINTF LPAREN STRING COMMA ID RPAREN SEMI
      {
          printf("  movl -%s, %%eax\n", $5);
          printf("  pushl %%eax\n");
          printf("  pushl $.LC0\n");
          printf("  call printf\n");
          printf("  addl $8, %%esp\n");
      }
    | RETURN NUM SEMI
      { /* handled in program rule */ }
    ;
idlist
    : ID
    | idlist COMMA ID
    ;
